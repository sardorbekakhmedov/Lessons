abstract class Vehicle  // Not new object
{
    public string Name;
    public string Year;
    public string Color;

    public abstract void Show();

}
class Car : Vehicle  //  Sealed not inheritance
{
    public string City;

    public  sealed override void Show()  // sealed
    {
        Console.WriteLine($" Car:  {City}");
    }
}
